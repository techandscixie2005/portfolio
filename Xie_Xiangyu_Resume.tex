\begin{cvsection}{Research Experience}
	\begin{cvsubsection}{Graduate Project}{Supervisor: Prof. Jun Jiang}{Fall 2025 -- Spring 2026}
		\textbf{Prediction of Room-Temperature Phosphorescence Molecular Properties Based on Equivariant Graph Neural Networks}
		\begin{itemize}
			\item \textbf{Dataset:} Curated 6,000 RTP-active organic molecules with DFT-calculated photophysical properties ($\Delta E_{S_1-T_1}$, SOC, oscillator strength)
			\item \textbf{Model Architecture:} Designed 8-layer EGNN\_Sparse\_Network with SE(3)-equivariant message passing; implemented 21-dim node features (atom type, hybridization, aromaticity) and 6-dim edge features
			\item \textbf{Training Strategy:} Applied geometric data augmentation (rotation, translation), layer-wise incremental Dropout, and ReduceLROnPlateau scheduling to enhance generalization under small-sample scenario
			\item \textbf{Results:} Achieved $R^2$=0.9855 for $SOC_{T_1-S_0}$ prediction and $R^2$=0.8781 for $\Delta E_{S_1-T_1}$ prediction on test set
			\item \textbf{Innovation:} First systematic application of equivariant GNN to RTP property prediction; proposed adaptive pooling strategy combining mean/max/sum pooling with learnable weights
			\item \textbf{Skills:} PyTorch, Graph Neural Networks, SE(3) Equivariance, Quantum Chemistry, Python, Linux
		\end{itemize}
	\end{cvsubsection}
